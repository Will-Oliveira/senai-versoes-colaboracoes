cd "nome da pasta"
git init
git status
git add "nome do arquivo"

git commit
:q
git commit -m "descrição do que foi feito no arquivo"
git config --global user.name "seu nome GitHub"

git log
git restore
git reset --hard HEAD~1 (~2 ~3...)
git branch "nome do branch"
git checkout "nome do branch"
git remote -v
git branch -M main
git branch
git push -u origin main
git push -u origin "nome da branch do desenvolvimento"
git checkout -b "nome da branch"
